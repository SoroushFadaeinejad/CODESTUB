// Example logic used inside a browser extension content script async function testCodesOnCheckout(codesToTest, inputSelector, applyBtnSelector) { let bestCode = null; let maxSavings = 0; for (const code of codesToTest) { const input = document.querySelector(inputSelector); const applyBtn = document.querySelector(applyBtnSelector); if (!input || !applyBtn) continue; // 1. Inject code into checkout input input.value = code; input.dispatchEvent(new Event('input', { bubbles: true })); // 2. Trigger the apply button applyBtn.click(); // 3. Wait for merchant server response await new Promise(resolve => setTimeout(resolve, 1500)); // 4. Read updated order subtotal from the page const currentSubtotal = parseSubtotalFromDOM(); const savings = calculateSavings(currentSubtotal); if (savings > maxSavings) { maxSavings = savings; bestCode = code; } } return { bestCode, maxSavings }; }
