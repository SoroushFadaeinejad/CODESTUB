async function getDealsForBrand(brandName) { try { const response = await fetch(`https://api.yourcouponprovider.com/v1/deals?brand=${encodeURIComponent(brandName)}`, { headers: { 'Authorization': 'Bearer YOUR_API_KEY' } }); const data = await response.json(); return data.coupons; // Returns live array of active codes } catch (error) { console.error("Failed to fetch live deals", error); return []; } }
